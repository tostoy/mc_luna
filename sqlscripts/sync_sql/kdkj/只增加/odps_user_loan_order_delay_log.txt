CREATE TABLE odps_user_loan_order_delay_log LIKE odps_user_loan_order_delay_log_tmp;

--- 将历史表中的数据根据数据的创建的时间插入目标表对应的分区中
INSERT OVERWRITE TABLE odps_user_loan_order_delay_log PARTITION (pt)
SELECT id, order_id, user_id, service_fee, counter_fee, late_fee, delay_day, principal, status, coupon_id, coupon_amount, remark, created_at, updated_at, repay_account_id, late_day, plan_fee_time, TO_CHAR(FROM_UNIXTIME(created_at), 'yyyymmdd') AS pt 
FROM odps_user_loan_order_delay_log_tmp;
